

USE CODESTATION;
CREATE TABLE USER(
	UID VARCHAR(30),
    UNAME VARCHAR(30),
    UMAIL VARCHAR(30) PRIMARY KEY,
    INSTITUTE VARCHAR(30),
    PASSWORD VARCHAR(30) 
);

CREATE TABLE CODECHEF(
	CODECHEFID VARCHAR(30) PRIMARY KEY,
    UMAIL VARCHAR(30),
    FOREIGN KEY (UMAIL) REFERENCES USER(UMAIL),
    RATING VARCHAR(30),
    FULL_SOV VARCHAR(30),
    PAR_SOV VARCHAR(30),
    GLO_RANK VARCHAR(30),
    COU_RANK VARCHAR(30)
);



CREATE TABLE CODEFORCES(
	CODEFORCES VARCHAR(30) PRIMARY KEY,
    UMAIL VARCHAR(30),
    FOREIGN KEY (UMAIL) REFERENCES USER(UMAIL),
    CUR_RATING VARCHAR(30),
    HIG_RATING VARCHAR(30),
    SOV_PROB VARCHAR(30)
);


CREATE TABLE INTERVIEWBIT(
	INTERVIEWBITID VARCHAR(30) PRIMARY KEY,
    UMAIL VARCHAR(30),
    FOREIGN KEY (UMAIL) REFERENCES USER(UMAIL),
    STREAK VARCHAR(30),
    SCORE VARCHAR(30),
    URANK VARCHAR(30)
    
);

CREATE TABLE GFG(
	GFGID VARCHAR(30) PRIMARY KEY,
    UMAIL VARCHAR(30),
    FOREIGN KEY (UMAIL) REFERENCES USER(UMAIL),
    SOLVED_PROB VARCHAR(30),
    CODING_SCOR VARCHAR(30),
    MON_STREAK VARCHAR(30)
);

CREATE TABLE CODESTATION(
    UMAIL VARCHAR(30),
    FOREIGN KEY (UMAIL) REFERENCES USER(UMAIL),
    SOL_PROB VARCHAR(30),
    OVERALL_SCORE VARCHAR(30)
	
);
USE CODESTATION;
CREATE TABLE ACTIVITY(
    UMAIL VARCHAR(30),
    FOREIGN KEY (UMAIL) REFERENCES USER(UMAIL),
	CC VARCHAR(30),
    CF VARCHAR(30),
    HE VARCHAR(30)
);

USE CODESTATION;
CREATE TABLE CONTESTS(
	CNAME VARCHAR(70) PRIMARY KEY,

    CDATE VARCHAR(30),

);
